// firestore.rules.txt — paste in Firestore → Rules → Publish
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // allow reading usernames before auth for login
    match /usernames/{doc} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    // optional: if you also use /users for pre-login checks
    match /users/{doc} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    // app data (requires sign-in)
    match /orders/{doc} { allow read, write: if request.auth != null; }
    match /drivers/{doc} { allow read, write: if request.auth != null; }
    match /clients/{doc} { allow read, write: if request.auth != null; }
    match /contracts/{doc} { allow read, write: if request.auth != null; }
    match /cod_settlements/{doc} { allow read, write: if request.auth != null; }
    match /payroll/{doc} { allow read, write: if request.auth != null; }
    match /pricing/{doc} { allow read, write: if request.auth != null; }
    match /fleet/{doc} { allow read, write: if request.auth != null; }
    match /settings/{document} { allow read, write: if request.auth != null; }
  }
}
